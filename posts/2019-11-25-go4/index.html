map

创建

  • var 变量名 map[key类型]value类型
  • 变量名 := map[key类型]value类型{}
  • make(map[key类型]value类型)
func main(){
	var m map[int]string
	m1 := map[int]string{}
	m2 := make(map[int]string)
}

初始化

func main(){
	var m /*map[int]string*/ = map[int]string{1:"张三", 2:"李四", 3:"王五"}

	m1 := map[int]string{1:"张三", 2:"李四", 3:"王五"}

	m2 := make(map[int]string)
	m2[1] = "张三"
	m2[2] = "李四"
}
//map[ 3:"王五" 1:"张三" 2:"李四"]

key/value

value,bool := map变量名[key]: 获取map中key的值，如果value存在，则bool为true，否则为false

func main(){
	var m = map[int]string{1:"张三", 2:"李四", 3:"王五"}
	value, ok := m[2]
	if ok {
		fmt.Println(value)
	}else {
		fmt.Println("not found")
	}
}

结构体

结构体是由一系列具有相同类型或不同类型的数据构成的数据集合

创建及初始化

type 结构体名 struct{}

func main(){
    type Student struct {
    //成员名称前不能添加var
        id int
        name string
        age int
    }
}

初始化方式：

  • 顺序初始化
  • 指定成员初始化
  • 通过"结构体变量.成员"方式初始化 (常用)
func main(){
    type Student struct {
    //成员名称前不能添加var
        id int
        name string
        age int
    }
    var s = Student{101, "张三", 18}
    fmt.Println(s) //{101 "张三" 18}
}

func main(){

结构体数组

type Student struct {
    id int
    name string
    age int
}
func main(){
    var arr = [3]Student{
        {101, "张三", 18},
        {102, "李四", 11},
        {103, "王五", 20},
    }
    fmt.Println(arr)
    //[{101 张三 18} {102 李四 11} {103 王五 20}]
}

//循环遍历
func main(){
    for key, value := range arr {
		fmt.Println(key, value)
	}
}
0 {101 张三 18}
1 {102 李四 11}
2 {103 王五 20}

指针

是一个特殊的变量，存储的是另一个变量的内存地址 var 变量名 *类型

func main(){
   var a = 10
	var p *int
	p = &a
	fmt.Printf("%p\n", &a) //0xc000088000
	fmt.Printf("%p\n", p)  //0xc000088000
 	fmt.Println(*p)   //10
 	//通过指针修改变量的值
	*p = 20
	fmt.Println(a)   //20
}

空指针

func main(){
    var p *int
    fmt.Println(p)   //<nil>
}

野指针

func main(){
    var p *int
	*p = 3
	fmt.Println(p)
	//panic: runtime error: invalid memory address or nil pointer dereference
}

new函数: 开辟数据类型对应的内存空间，返回值为数据类型指针

func main(){
    var p *int
    p = new(int)
    *p  = 2
	 fmt.Println(*p) //2
}

指针作为函数参数

func update(p *int){
	*p = 1
}
func main(){
	var num int = 10

	update(&num)
	fmt.Println(num) //1
}

数组指针

定义数组指针 : var 变量名 *[长度]类型

func main(){
    nums := [3]int{1,2,3}
    var p *[3]int
    p = &nums
    fmt.Println(*p)      //[1 2 3]
    fmt.Println((*p)[0]) //1
    //[]的运算优先级高于*
    fmt.Println(p[0])    //1  **切片中不能使用**

    updateArr(p)
}

//数组指针作为参数
func updateArr(p *[10]int){
    p[0] = 10
}

指针数组

是数组中存储的都是指针; 数组元素是指针 var 变量名 [长度]*类型

func main(){
    var p[2]*int
    var i = 10
    var j = 20

    p[0] = &i
    p[1] = &j
    fmt.Println(p)     //[0xc000018040 0xc000018048]
}
